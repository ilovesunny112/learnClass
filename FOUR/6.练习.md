### 有效数字
```javascript
// 正数 负数 11 -11
// 整数 小数 0 12 0.2 
// 只出现小数点，后面至少要跟一位数字
// 小数点前面必须有数字
/^(+|-)?(\d|([1-9]\d+))(\.\d+)?$/.test()
```

### 邮箱
```javascript
// \w+
// ( (-\w+) |  (\.\w+) )*
// @
// [A-Za-z0-9]+
// ((\.\-)[A-Za-z0-9]+)*
// \.
// [A-Za-z0-9]+

var reg = /^\w+((-\w+)|(\.\w+))*@[A-Za-z0-9]+((\.\-)[A-Za-z0-9]+)*\.[A-Za-z0-9]+$/
```
